### Step 2.2: Create `QA_API__Session` (browser lifecycle per request)

```python
# sg_send_qa/api/QA_API__Session.py
from osbot_utils.type_safe.Type_Safe import Type_Safe
from sg_send_qa.api.Schema__QA_Request import Schema__QA_Request

class QA_API__Session(Type_Safe):
    request : Schema__QA_Request = None

    def __enter__(self):
        if self.request.target == 'local':
            from sg_send_qa.browser.SG_Send__Browser__Test_Harness import SG_Send__Browser__Test_Harness
            self.harness = SG_Send__Browser__Test_Harness().headless(True).setup()
            self.sg_send = self.harness.sg_send
            self.access_token = self.harness.access_token()
        else:
            # live mode: browser pointed directly at target URL
            # TODO: wire up live mode (no harness, no local servers)
            raise NotImplementedError("live mode not yet implemented")
        return self

    def __exit__(self, *args):
        if hasattr(self, 'harness'):
            self.harness.teardown()
```

### Step 2.3: Create `QA_API__Runner` (execution engine)

```python
# sg_send_qa/api/QA_API__Runner.py
import time, uuid
from osbot_utils.type_safe.Type_Safe           import Type_Safe
from sg_send_qa.api.Schema__QA_Request         import Schema__QA_Request
from sg_send_qa.api.Schema__QA_Response        import Schema__QA_Response
from sg_send_qa.api.Schema__Capture_Config     import Schema__Capture_Config
from sg_send_qa.api.QA_API__Session            import QA_API__Session

class QA_API__Runner(Type_Safe):
    request : Schema__QA_Request = None

    def _ensure_capture(self):
        if self.request.capture is None:
            self.request.capture = Schema__Capture_Config()

    def _ensure_trace_id(self):
        if not self.request.trace_id:
            self.request.trace_id = str(uuid.uuid4())[:8]

    def run(self, workflow_fn) -> dict:
        self._ensure_capture()
        self._ensure_trace_id()
        start    = time.time()
        response = Schema__QA_Response(trace_id=self.request.trace_id)
        try:
            with QA_API__Session(request=self.request) as session:
                result = workflow_fn(session)
                response.status = 'pass'
                response.duration_ms = int((time.time() - start) * 1000)
                return {**response.json(), **result}
        except Exception as e:
            response.status     = 'fail'
            response.error      = str(e)
            response.duration_ms = int((time.time() - start) * 1000)
            return response.json()
```

## Task 3: v030 Migration — Concrete Steps

### Phase 0: TestCase conversion (no infra change)

**Important note from harness code:** `SG_Send__Browser__Test_Harness.set_access_token()` already uses `with self.sg_send as _:` internally. In `setUpClass`, call it directly:

```python
cls.harness = SG_Send__Browser__Test_Harness().headless(True).setup()
cls.sg_send = cls.harness.sg_send
cls.harness.set_access_token()   # ← call on harness, not wrapped in with
```

**Before (current `p0__upload__single_file`):**
```python
class TestSingleFileUpload:
    def test_upload_page_loads(self, page, ui_url, screenshots):
        goto(page, f"{ui_url}/en-gb/")
        handle_access_gate(page, send_server.access_token)
        assert page.locator("upload-step-select").is_visible()

    def test_single_file_upload_flow(self, page, ui_url, send_server, screenshots):
        goto(page, f"{ui_url}/en-gb/")
        handle_access_gate(page, send_server.access_token)
        page.locator("#file-input").set_input_files(...)
        wait_for_upload_states(page, ["file-ready", "choosing-delivery"])
        ...
```

**After (TestCase + harness):**
```python
from unittest import TestCase
from sg_send_qa.browser.SG_Send__Browser__Test_Harness import SG_Send__Browser__Test_Harness

class test_Single_File_Upload(TestCase):

    @classmethod
    def setUpClass(cls):
        cls.harness = SG_Send__Browser__Test_Harness().headless(True).setup()
        cls.sg_send = cls.harness.sg_send
        cls.harness.set_access_token()

    @classmethod
    def tearDownClass(cls):
        cls.harness.teardown()

    def test__01__upload_page_loads(self):
        self.sg_send.page__root()
        assert self.sg_send.is_upload_zone_visible() or self.sg_send.is_access_gate_visible()

    def test__02__single_file_upload_flow(self):
        self.sg_send.page__root()
        self.sg_send.upload__set_file("test.txt", b"Hello from QA")
        self.sg_send.upload__click_next()
        self.sg_send.upload__select_share_mode("combined")
        self.sg_send.upload__click_next()
        self.sg_send.upload__wait_for_complete()
        link = self.sg_send.upload__get_combined_link()
        assert link and '#' in link
```

Key changes:
- `goto()` + `handle_access_gate()` → `sg_send.page__root()` (harness pre-injected the token)
- `page.locator(...)` → `sg_send.upload__*()` methods
- No fixture parameters — all state on `cls`
- Class name `test_*` (lowercase) to match integration test convention
- Tests numbered `test__01__`, `test__02__` for execution order

## Risks and Unknowns

| Risk | Mitigation |
|------|-----------|
| `#file-name` selector in `upload-step-select` shadow DOM — not confirmed | Run `page.locator("upload-step-select #file-name").text_content()` in debug mode against live UI |
